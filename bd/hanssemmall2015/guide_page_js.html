공통 JS적용 가이드 (2015.10.21)

"ui.plugins.js" 의 구현된 기능들 입니다. 실행되는 시점은 하단의 "ui.common.js" 에서 실행 된다.
해당 Plug-in 들의 세부 기능들을 사용하기 위해서는, 하단의 "ui.common.js" 다음에 실행하거나 "$(document).ready()" 시점에 실행하면 된다

1. SelectBox

> html

<!-- "ui-select-box" 공통 적용시 필수 클래스, width값을 변경하려면 !important; 로 width를 고정시킨다. -->
<div class="select_area ui-select-box">
   <!-- "ui-input-select" 필수 클래스 (select 자체를 비활성화 하기 위해서는 select에 disabled="disabled") -->
   <select class="ui-input-select">
      <option value="" selected="selected">옵션을 선택하세요</option>
      <option value="value2">옵션 Label 1</option>
      <option value="value3">옵션 Label 2</option>
      <option value="value4">옵션 Label 3</option>
      <option value="value5" disabled="disabled">옵션 Label 5</option>
   </select>
</div>

> Javascript

//기본적용 (ui.common.js 에서 실행해 주고 있다. 동적으로 생성한 selectbox 에 스킨을 적용하기 위해서 실행해주면 된다.)
$( target ).selectBox();

//option을 동적으로 추가해 주거나, selected, disabled 속성을 강제로 변경시 상태를 스킨에 반영하기 위해서 아래 방법을 사용한다.
$( target ).trigger( 'selectbox:updated' );

//change 이벤트 등록 (대상 select 태그에 이벤트를 등록한다.)
$( selectbox ).on( 'change', function (e) {
    //
});

3. NumberCounter

> html

<!-- "ui-number-counter" 필수 클래스 -->
<span class="opt_spinner ui-number-counter">
   <!-- "ui-decrease" 필수 클래스 -->
   <button class="opt_down ui-decrease">수량 줄이기</button>
   <!-- "ui-number-input" 필수 클래스 (min=최소값 max=최대값, value=현재값), input에만 disabled="disabled" 을 적용하면 버튼까지 disabled 속성이 적용된다.-->
   <input class="ui-number-input" type="text" title="수량입력" value="1" min="1" max="20" readonly="readonly" />
   <!-- "ui-increase" 필수 클래스 -->
   <button class="opt_up ui-increase">수량 더하기</button>
</span>

> Javascript

//기본적용 (ui.common.js 에서 실행해 주고 있다. 동적으로 생성한 numberCounter 에 기능을 적용하기 위해서 실행해주면 된다.)
$( target ).numberCounter();

//value의 값을 강제로 변경할때 사용한다. (value가 min, max 영역을 벗어나면 value 최대 최소치 구간으로 변경한다.)
$( target ).trigger( 'numbercounter:change_value', value );

//change 이벤트 등록 (대상 input 태그에 이벤트를 등록한다.)
$( input ).on( 'change', function (e) {
    //e.currentTarget.value
});

4. TAB

  • tab1
  • tab2
컨텐츠1
컨텐츠2

> html

<!-- "ui-tab-area" 필수 클래스 -->
<div class="tab_area ui-tab-area">
   <!-- "ui-tabs" 필수 클래스 -->
   <ul class="ui-tabs tab_tit">
      <!-- "ui-tab-item" 필수 클래스 -->
      <li class="ui-tab-item"><a href="#">tab1</a></li>
      <li class="ui-tab-item"><a href="#">tab2</a></li>
   </ul>
   <!-- "ui-tab-contents" 필수 클래스 (탭을 클릭시 콘텐츠 영역 동기화를 시키지 않으려면 "ui-tab-contents" class를 넣지 않으면 된다) -->
   <div class="ui-tab-contents tab_body">
      <!-- "ui-tab-contents-item" 필수 클래스 -->
      <div class="ui-tab-contents-item tab-cont">
         컨텐츠1
      </div>
      <div class="ui-tab-contents-item tab-con">
         컨텐츠2
      </div>
   </div>
</div>

> Javascript

//기본적용 (ui.common.js 에서 실행해 주고 있다. 동적으로 생성한 tab 에 기능을 적용하기 위해서 실행해주면 된다.)
$( target ).tabs({
    eventBlock: false //true를 설정하면 클릭이벤트 전달을 막는다. (a href=의 동작을 막는다), 기본설정=true
});

//현재 선택된 Tab 변경 (탭의 갯수보다 큰 index를 설정하면 동작하지 않는다.)
$( target ).trigger( 'tabs:change_index', 2 );

//이벤트 등록
$( target ).on( 'tabs:change', function ( e, index ) {
    console.log( index ); // 현재 선택된 탭의 index
});

LayerPopup 띄우기

개인구매회원 혜택

팝업 컨텐츠

> html

<div>
   <!-- a 태그에서 실행 -->
   <a href="javascript:$('.layer_popup').openLayer();">LayerPopup 띄우기</a>
   <!-- button 에서 실행 -->
   <button type="button" class="botton_st_03 purple" onclick="$('.layer_popup').openLayer();">LayerPopup 띄우기</button>
</div>
<!-- layer_popup -->
<div class="layer_popup" style="display:none;">
   <div class="layer_pop_head">
      <h1 class="ui-layer-title layer_title"><span class="point">개인</span>구매회원 혜택</h1>
      <!-- "ui-layer-close" 필수 클래스 -->
      <a href="#" class="ui-layer-close layer_close_btn"><img src="common/images/common/btn_pop_close.png" alt="닫기"></a>
   </div>
   <div class="layer_pop_cont">
      팝업 컨텐츠
   </div>
</div>

> Javascript

//Layer Popup 열기 ("X" 닫기 버튼의 닫기 이벤트는 자동 적용 된다.)
$( target ).openLayer();

//Layer Popup 닫기
$( target ).closeLayer();

//open, close 이벤트 등록
$( target ).on( 'layer:open layer:close', function (e) {
    //e.type - layer:open or layer:close
});
